	% 3. Bachelor Thesis
	\section*{BACHELOR’S THESIS}
	\textbf{Densification of Natural Wood to Improve Structural Properties} \\
	\textbf{Advisor:} Major Md.\ Anisur Rahman, EME, Assistant Professor, ME Dept., MIST
	
	\begin{itemize}
		\item Investigated the effects of three distinct processing conditions—natural (untreated), seasoned, and chemically treated with subsequent densification—on the structural performance of natural wood.
		\item Studied three tropical wood species: \textit{Swietenia macrophylla}, \textit{Albizia procera}, and \textit{Cordia subcordata}.
		\item Seasoned selected specimens under controlled conditions prior to testing, while chemically treated samples were pretreated using NaOH and Na$_2$SO$_3$ followed by thermo-mechanical hot-press densification.
		\item Conducted tensile, compression, three-point bending, impact, moisture absorption, and SEM analyses following ASTM standards.
		\item Performed a comparative evaluation across all three conditions; chemically treated and densified specimens exhibited the highest strength, while seasoned wood showed moderate improvements over natural samples.\\
	\end{itemize}
