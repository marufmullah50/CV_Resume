	% 3. Bachelor Thesis
	\section*{BACHELOR’S THESIS}
	\textbf{Densification of Natural Wood to Improve Structural Properties} \\
	\textbf{Advisor:} Major Md.\ Anisur Rahman, EME, Assistant Professor, ME Dept., MIST
	
	\begin{itemize}
		\item Studied structural performance of natural, seasoned, and chemically treated–densified wood.
		\item Investigated three tropical species: \textit{Swietenia macrophylla}, \textit{Albizia procera}, and \textit{Cordia subcordata}.
		\item Chemical treatment involved NaOH and Na$_2$SO$_3$ followed by thermo-mechanical hot-press densification.
		\item Conducted mechanical, moisture absorption, and SEM analyses following ASTM standards; densified samples showed highest strength.
	\end{itemize}
